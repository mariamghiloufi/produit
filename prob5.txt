ALGORITHM ortog


PROCEDURE prodscal(t1,t2,s=0)


FOR i FROM 1 to n step 1 do
FOR j FROM 1 to n step 1 do

s = s + t1[i]*t2[i];

END_FOR
END_FOR

RETURN sum;
END

VAR
    arr1, arr2 : arrayof INTEGER[10];
    i,j,n, sum : INTEGER;
   
BEGIN
write ("donner le nombre d'elements des vecteurs");
Read (n);


FOR i FROM 1 to n step 1 do
read(arr1[i]);
END_FOR


FOR i FROM 1 to n step 1 do
read(arr2[i]);
END_FOR

prodscal(arr1,arr2,sum);


IF (sum = 0) THEN
 write("les deux vecteurs sont orthogonaux");
END_IF

END
